## Context

**Code pattern:**
```typescript
import type { WeekRecord } from '../hooks/useAllWeeks';
import { getWeekStart, getYearFromWeekId } from './weekUtils';

/**
 * Get month (0-11) for a week using ISO Thursday rule.
 * Week belongs to the month containing its Thursday.
 */
export function getMonthFromWeekId(weekId: string): number {
  const weekStart = getWeekStart(weekId);
  const thursday = new Date(weekStart);
  thursday.setDate(thursday.getDate() + 3); // Mon + 3 = Thu
  return thursday.getMonth(); // 0-11
}

export function groupWeeksByYearAndMonth(
  weeks: WeekRecord[]
): Map<number, Map<number, WeekRecord[]>> {
  const groups = new Map();
  weeks.forEach(week => {
    const year = getYearFromWeekId(week.weekId);
    const month = getMonthFromWeekId(week.weekId);

    if (!groups.has(year)) groups.set(year, new Map());
    const yearMap = groups.get(year)!;

    if (!yearMap.has(month)) yearMap.set(month, []);
    yearMap.get(month)!.push(week);
  });
  return groups;
}

export function getNormalWeeks(weeks: WeekRecord[]): WeekRecord[] {
  return weeks.filter(week => week.status === 'normal');
}
```

**Interface:**
```typescript
export interface TimelineData {
  currentWeekId: string;
  availableYears: number[];  // Sorted newest first (2026, 2025, 2024...)
  getYearWeeks: (year: number) => WeekRecord[];
  getMonthWeeks: (year: number, month: number) => WeekRecord[];
  yearMonthGroups: Map<number, Map<number, WeekRecord[]>>;
}

export function useTimelineData(): TimelineData
```

**Implementation:**
- Call useAllWeeks() to get all weeks (already lazy loads)
- Use useMemo to group weeks client-side (instant for <1000 weeks)
- Extract availableYears, sort descending (newest first)
- Provide getter functions for year/month data
- Return currentWeekId for pinned week display

**Dependencies:**
- Import useAllWeeks, WeekRecord from ./useAllWeeks
- Import getCurrentWeekId from ../lib/weekUtils
- Import groupWeeksByYearAndMonth from ../lib/timelineUtils

**Code pattern:**
```typescript
import { useMemo } from 'react';
import { useAllWeeks, type WeekRecord } from './useAllWeeks';
import { getCurrentWeekId } from '../lib/weekUtils';
import { groupWeeksByYearAndMonth } from '../lib/timelineUtils';

export interface TimelineData {
  currentWeekId: string;
  availableYears: number[];
  getYearWeeks: (year: number) => WeekRecord[];
  getMonthWeeks: (year: number, month: number) => WeekRecord[];
  yearMonthGroups: Map<number, Map<number, WeekRecord[]>>;
}

export function useTimelineData(): TimelineData {
  const { weeks } = useAllWeeks();
  const currentWeekId = getCurrentWeekId();

  const yearMonthGroups = useMemo(
    () => groupWeeksByYearAndMonth(weeks),
    [weeks]
  );

  const availableYears = useMemo(
    () => Array.from(yearMonthGroups.keys()).sort((a, b) => b - a), // Newest first
    [yearMonthGroups]
  );

  const getYearWeeks = useMemo(
    () => (year: number): WeekRecord[] => {
      const monthMap = yearMonthGroups.get(year);
      if (!monthMap) return [];
      return Array.from(monthMap.values()).flat();
    },
    [yearMonthGroups]
  );

  const getMonthWeeks = useMemo(
    () => (year: number, month: number): WeekRecord[] => {
      return yearMonthGroups.get(year)?.get(month) || [];
    },
    [yearMonthGroups]
  );

  return {
    currentWeekId,
    availableYears,
    getYearWeeks,
    getMonthWeeks,
    yearMonthGroups,
  };
}
```

**Notes:**
- useMemo depends on [weeks] array to auto-recalculate on data changes
- getYearWeeks flattens all month arrays for that year
- getMonthWeeks returns specific month array or empty array if not found
- No Firebase queries here - grouping is client-side transformation
